📄 Jenkinsfile Structure

pipeline {
    agent any
    
    stages {
        stage('Checkout') {
            steps {
                checkout scm
            }
        }
        stage('Build') {
            steps {
                sh 'mvn clean package'
            }
        }
        stage('Test') {
            steps {
                sh 'mvn test'
            }
        }
        stage('Deploy') {
            steps {
                sh 'docker build && docker push'
            }
        }
    }
    
    post {
        failure {
            slackSend(message: "Build Failed!")
        }
    }
}

🐳 Docker for Microservices

Dockerfile Best Practices

# Multi-stage build
FROM maven:3.9-eclipse-temurin-17 AS builder
WORKDIR /app
COPY pom.xml .
RUN mvn dependency:go-offline
COPY src ./src
RUN mvn clean package

# Runtime stage
FROM eclipse-temurin:17-jre-alpine
RUN addgroup -S app && adduser -S app -G app
COPY --from=builder /app/target/*.jar app.jar
USER app
EXPOSE 8081
HEALTHCHECK --interval=30s CMD curl -f health
ENTRYPOINT ["java", "-jar", "app.jar"]

📦 Docker Compose

version: '3.8'
services:
  user-service:
    build: ./user-service
    ports:
      - "8081:8081"
    healthcheck:
      test: ["CMD", "curl", "-f", "http://localhost:8081/health"]
      interval: 30s
      retries: 3
    networks:
      - microservices
      
  order-service:
    build: ./order-service
    ports:
      - "8082:8082"
    depends_on:
      user-service:
        condition: service_healthy
        
networks:
  microservices:
    driver: bridge

🚨 Alerting & Notifications

Notification Channels

  • Slack - Real-time team alerts
  • Email - Detailed reports
  • PagerDuty - Critical escalation
  • Teams - Microsoft integration

Alert Triggers

post {
    success {
        slackSend("✅ Build Passed")
    }
    failure {
        slackSend("❌ Build Failed!")
        emailext("ALERT: Build Failed")
    }
}

📊 Alert Configuration

NOTIFICATIONS:
  SLACK:
    enabled: true
    channel: "#cicd-alerts"
    events:
      - build_failure      # Critical
      - build_success       # Info
      - security_scan_fail  # Critical
      - deployment_failure  # High
      
  EMAIL:
    enabled: true
    recipients:
      - devops-team@cdac.com
      - developers@cdac.com
    attachLog: true
        
ESCALATION:
  build_failure:
    level 1 → Developers (0 min)
    level 2 → DevOps Lead (15 min)  
    level 3 → On-Call (30 min)

⌨️ Key Commands

# Start local environment
docker-compose up -d

# Run specific service
docker-compose up user-service

# View logs
docker-compose logs -f order-service

# Build Docker image
docker build -t user-service:latest ./user-service

# Run pipeline manually
curl -X POST JENKINS_URL/job/pipeline/build

# Check container health
docker inspect --format='{{.State.Health.Status}}' user-service

# Security scan
trivy image user-service:latest
