📄 GitHub Actions Workflow

name: DevSecOps CI/CD Pipeline

on:
  push:
    branches: [ main, develop ]
  pull_request:
    branches: [ main ]

jobs:
  build:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-java@v4
        with:
          java-version: '17'
      - run: mvn clean package
      - run: mvn test
      
  security:
    runs-on: ubuntu-latest
    steps:
      - uses: aquasecurity/trivy-action@master
        with:
          scan-type: 'fs'

⚙️ Jenkinsfile Structure

pipeline {
    agent any
    
    stages {
        stage('Checkout') {
            steps { checkout scm }
        }
        stage('Build') {
            steps { sh 'mvn clean package' }
        }
        stage('Test') {
            steps { sh 'mvn test' }
        }
        stage('Docker Build') {
            steps { sh 'docker build -t app .' }
        }
    }
    
    post {
        failure {
            slackSend(message: "❌ Build Failed!")
        }
    }
}

🐳 Docker for Microservices

Dockerfile Best Practices

# Multi-stage build
FROM maven:3.9-eclipse-temurin-17 AS builder
WORKDIR /app
COPY pom.xml .
RUN mvn dependency:go-offline
COPY src ./src
RUN mvn clean package

# Runtime stage - minimal image
FROM eclipse-temurin:17-jre-alpine
RUN addgroup -S app && adduser -S app -G app
COPY --from=builder /app/target/*.jar app.jar
USER app
EXPOSE 8081
HEALTHCHECK --interval=30s CMD wget --spider localhost:8081/health
ENTRYPOINT ["java", "-jar", "-Xms256m", "-Xmx512m", "app.jar"]

📦 Docker Compose

version: '3.8'
services:
  user-service:
    build: ./user-service
    ports:
      - "8081:8081"
    healthcheck:
      test: ["CMD", "curl", "-f", "http://localhost:8081/health"]
      interval: 30s
      retries: 3
    networks:
      - microservices
      
  order-service:
    build: ./order-service
    ports:
      - "8082:8082"
    depends_on:
      user-service:
        condition: service_healthy
        
networks:
  microservices:
    driver: bridge

Notification Channels

  • Slack - Real-time alerts

GitHub Actions Alert

- name: Slack Alert
  uses: slackapi/slack-github-action
  with:
    payload: |
      {"text": "❌ Build Failed!"}
  env:
    SLACK_WEBHOOK_URL: ${{ secrets.SLACK_WEBHOOK_URL }}

📊 Alert Configuration
